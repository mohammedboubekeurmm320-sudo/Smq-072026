  1174→-- TRIGGER — Audit trail auto-logging for critical tables
  1175→-- ============================================================================
  1176→
  1177→CREATE OR REPLACE FUNCTION log_audit_trail()
  1178→RETURNS TRIGGER AS $$
  1179→DECLARE
  1180→  v_audit_action text;
  1181→  v_org_id uuid;
  1182→BEGIN
  1183→  IF (TG_OP = 'INSERT') THEN
  1184→    v_audit_action := 'CREATE';
  1185→    v_org_id := NEW.organization_id;
  1186→    INSERT INTO audit_trails (audit_action, table_name, record_id, new_values, organization_id)
  1187→    VALUES (v_audit_action, TG_TABLE_NAME, NEW.id, to_jsonb(NEW), v_org_id);
  1188→    RETURN NEW;
  1189→  ELSIF (TG_OP = 'UPDATE') THEN
  1190→    v_audit_action := 'UPDATE';
  1191→    v_org_id := NEW.organization_id;
  1192→    INSERT INTO audit_trails (audit_action, table_name, record_id, old_values, new_values, organization_id)
  1193→    VALUES (v_audit_action, TG_TABLE_NAME, NEW.id, to_jsonb(OLD), to_jsonb(NEW), v_org_id);
  1194→    RETURN NEW;
  1195→  ELSIF (TG_OP = 'DELETE') THEN
  1196→    v_audit_action := 'DELETE';
  1197→    v_org_id := OLD.organization_id;
  1198→    INSERT INTO audit_trails (audit_action, table_name, record_id, old_values, organization_id)
  1199→    VALUES (v_audit_action, TG_TABLE_NAME, OLD.id, to_jsonb(OLD), v_org_id);
  1200→    RETURN OLD;
  1201→  END IF;
  1202→END;
  1203→$$ LANGUAGE plpgsql SECURITY DEFINER;
