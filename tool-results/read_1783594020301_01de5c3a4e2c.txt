   268→  /** Layer 1: Transition FormTemplate status with full role-based guard and e-signature enforcement */
   269→  transitionFormTemplateStatus: (templateId, targetStatus, userId, userRole, signatureHash, comment) => {
   270→    const state = get();
   271→    const template = state.formTemplates.find(t => t.id === templateId);
   272→    if (!template) return { success: false, error: 'Template not found' };
   273→
   274→    // Validate transition is allowed
   275→    const currentStatus = template.status || (template.isActive ? 'Approved' : 'Draft');
   276→    const allowedTransitions = FORM_TEMPLATE_TRANSITIONS[currentStatus];
   277→    if (!allowedTransitions || !allowedTransitions.includes(targetStatus)) {
   278→      return { success: false, error: `Transition from ${template.status} to ${targetStatus} is not allowed` };
   279→    }
   280→
   281→    // Validate role is authorized for this transition
   282→    const transitionKey = `${currentStatus}→${targetStatus}`;
   283→    const allowedRoles = FORM_TEMPLATE_TRANSITION_ROLES[transitionKey];
   284→    if (!allowedRoles || !allowedRoles.includes(userRole)) {
   285→      return { success: false, error: `Role '${userRole}' is not authorized for transition ${transitionKey}` };
   286→    }
   287→
   288→    // For approval (Under_Review → Approved), enforce e-signature
   289→    const requiresESig = targetStatus === 'Approved';
   290→    if (requiresESig && !signatureHash) {
   291→      return { success: false, error: 'Electronic signature is required for template approval' };
   292→    }
   293→
   294→    // Build signature record if provided
   295→    const newSignature: ElectronicSignature | undefined = signatureHash ? {
   296→      id: `sig-${Date.now()}-${Math.random().toString(36).substring(2, 7)}`,
   297→      documentId: templateId,
   298→      signedById: userId,
   299→      signerName: state.profiles.find(p => p.id === userId)?.fullName || userId,
   300→      signerRole: userRole,
   301→      signatureType: targetStatus === 'Approved' ? 'approval' : targetStatus === 'Obsolete' ? 'rejection' : 'review',
   302→      signatureHash,
   303→      revoked: false,
   304→      createdAt: new Date().toISOString(),
   305→    } : undefined;
   306→
   307→    // Compute updated fields
   308→    const now = new Date().toISOString();
   309→    const updates: Partial<FormTemplate> = {
   310→      status: targetStatus,
   311→      isActive: targetStatus === 'Approved', // isActive derived from status
   312→      updatedAt: now,
   313→      reviewComment: comment,
   314→    };
   315→    if (targetStatus === 'Approved') {
   316→      updates.effectiveDate = now;
   317→    }
   318→    if (newSignature) {
   319→      updates.signatures = [...(template.signatures || []), newSignature];
   320→    }
   321→
   322→    // Apply update
   323→    set(st => ({
   324→      formTemplates: st.formTemplates.map(t => t.id === templateId ? { ...t, ...updates } : t),
   325→    }));
   326→
   327→    // Log audit trail
   328→    const updatedTemplate = get().formTemplates.find(t => t.id === templateId);
   329→    get().logAudit(
   330→      targetStatus === 'Approved' ? 'APPROVE' : targetStatus === 'Obsolete' ? 'REJECT' : 'UPDATE',
   331→      'FormTemplate',
   332→      templateId,
   333→      { status: template.status },
   334→      { status: targetStatus, approvedBy: userId, comment }
   335→    );
   336→
   337→    return { success: true };
   338→  },
